// clase Producto

Clase Producto (abstracta)
    Atributos:
        nombre (String)
        precio (double)
    
    Métodos:
        Constructor(nombre, precio)
        imprimirDetalles() (método abstracto)

// clase Libro

Clase Libro (extiende Producto, implementa Describible)
    Atributos:
        autor (String)

    Metodos:
        Constructor(nombre, precio, autor)
        imprimirDetalles():
            Mostrar "Titulo: " + nombre
            Mostrar "Precio: " + precio
            Mostrar "Autor: " + autor
        describir():
            Mostrar "Descripcion del libro: " + nombre + " por " + autor

// clase Revista

Clase Revista (extiende Producto)
    Atributos:
        edicion (String)

    Metodos:
        Constructor(nombre, precio, edicion)
        imprimirDetalles():
            Mostrar "Titulo: " + nombre
            Mostrar "Precio: " + precio
            Mostrar "Edicion: " + edicion

// interfaz Describible

Interfaz Describible
    Metodo describir()

// clase Biblioteca

Clase Biblioteca
    Atributo estatico:
        contadorProductos (int)

    Metodos:
        Metodo estatico contarProductos():
            Retornar contadorProductos
        Metodo estatico incrementarContador():
            contadorProductos += 1

// function menu, para hacerlo fuera del main

Funcion menu():
    Inicializar variable opcion

    Hacer mientras no salir:
        Mostrar "1. Agregar libro"
        Mostrar "2. Agregar revista"
        Mostrar "3. Salir"
        Leer opcion

        Si opcion == 1:
            Leer nombre, precio y autor

	    Hacer mientras precio < 0:
                Mostrar "El precio no puede ser negativo. Ingrese un precio valido:"
                Leer precio

            Crear un nuevo Libro(nombre, precio, autor)
            Biblioteca.incrementarContador()
            Mostrar detalles del libro
        
        Si opcion == 2:
            Leer nombre, precio y edicion

            Hacer mientras precio < 0:
                Mostrar "El precio no puede ser negativo. Ingrese un precio valido:"
                Leer precio

            Crear una nueva Revista(nombre, precio, edicion)
            Biblioteca.incrementarContador()
            Mostrar detalles de la revista
        
        Si opcion == 3:
            Mostrar "Saliendo"
	
	default:
	    Mostrar "Opcion no valida. Intente de nuevo."

    Mientras opcion != 3

    Mostrar "Cantidad total de productos ingresados: " + Biblioteca.contarProductos()

// Main

Funcion main()
    Llamar a menu()
